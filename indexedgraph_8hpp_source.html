111 register_edge_type(const E_TYPE &edge) noexcept {
112 std::size_t attempt_id{};
113 std::size_t id;
114 while (attempt_id < std::numeric_limits<std::size_t>::max()) {
115 auto hash1{std::hash<E_TYPE>()(edge)};
116 id = hash1 ^ (attempt_id + (hash1 << 6) + (hash1 >> 2) + 0x9e3779b9);
117 if (edge_index.find(id) == edge_index.end()) {
118 edge_index[id] = edge;
119 } else if (edge_index.at(id) == edge) {
120 return thesoup::types::Result<std::size_t, IndexedPropertyDiGraphAttributes::ErrorCode>::success(
121 id);
122 } else {
123 attempt_id++;
124 }
125 }
126
127 return thesoup::types::Result<std::size_t, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(
128 IndexedPropertyDiGraphAttributes::ErrorCode::TOO_MANY_DUPLICATE_IDS);

154 std::size_t attempt_id {};
155 std::size_t id;
156 while (attempt_id < std::numeric_limits<std::size_t>::max()) {

186 if (adj_list.find(vertex) == adj_list.end()) {
187 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(
188 IndexedPropertyDiGraphAttributes::ErrorCode::NON_EXISTENT_VERTEX);
189 } else if (adj_list.at(vertex).size() > 0 || vertex_index.at(vertex).incoming_edges > 0) {
190 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(
191 IndexedPropertyDiGraphAttributes::ErrorCode::CONNECTED_VERTEX);
192 } else {
193 adj_list.erase(vertex);
194 vertex_index.erase(vertex);
195 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::success(
196 thesoup::types::Unit::unit);

231 insert_edge(const thesoup::types::Edge<std::size_t, std::size_t> &edge) noexcept {
232 if (vertex_index.find(edge.from) == vertex_index.end() ||
233 vertex_index.find(edge.to) == vertex_index.end()) {
234 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(

239 } else {
240 adj_list.at(edge.from).insert({edge.edge_type, edge.to});
241 vertex_index.at(edge.to).incoming_edges++;
242 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::success(
243 thesoup::types::Unit::unit);

257 thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>
258 delete_edge(const thesoup::types::Edge<std::size_t, std::size_t>& edge) noexcept {
259 std::size_t from_id {edge.from};
260 std::size_t to_id {edge.to};
261 std::size_t edge_type_id {edge.edge_type};
262
263 if (adj_list.find(from_id) == adj_list.end() || adj_list.find(to_id) == adj_list.end()) {
264 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(
265 IndexedPropertyDiGraphAttributes::ErrorCode::NON_EXISTENT_VERTEX);
266 } else if (edge_index.find(edge_type_id) == edge_index.end()) {
267 return thesoup::types::Result<thesoup::types::Unit, IndexedPropertyDiGraphAttributes::ErrorCode>::failure(
268 IndexedPropertyDiGraphAttributes::ErrorCode::INVALID_EDGE_TYPE);
269 } else {
270 auto it{adj_list.at(from_id).find({edge_type_id, to_id})};
271 if (it == adj_list.at(from_id).end()) {
272 return thesoup::types::Result < thesoup::types::Unit,
273 IndexedPropertyDiGraphAttributes::ErrorCode > ::failure(
274 IndexedPropertyDiGraphAttributes::ErrorCode::NON_EXISTENT_EDGE);
275 } else {
276 adj_list.at(from_id).erase(it);
277 vertex_index.at(to_id).incoming_edges--;
278 return thesoup::types::Result < thesoup::types::Unit,
279 IndexedPropertyDiGraphAttributes::ErrorCode > ::success(
280 thesoup::types::Unit::unit);
