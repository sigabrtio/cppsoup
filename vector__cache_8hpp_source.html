201 void push_back(const T& item) noexcept {
202 std::size_t page_number_to_insert_to {(_size & page_number_bit_mask) >> page_offset_bits};
203 std::size_t page_offset {_size & page_offset_bit_mask};
204 std::size_t page_table_entry {page_number_to_insert_to & page_table_entry_bit_mask};
205
206 // Page swap is needed
207 if (page_table[page_table_entry].page_number != page_number_to_insert_to || page_number_to_insert_to >= num_pages) {
208
209 if (page_table[page_table_entry].valid) {
210 saver(page_table[page_table_entry].slice, page_table[page_table_entry].page_number);
211 }
212 // New page needed
213 if (page_number_to_insert_to >= num_pages) {
214 T* new_page = new T[page_size];
215 page_table[page_table_entry].slice.start = new_page;
216 page_table[page_table_entry].slice.size = 0;
217 num_pages++;
218 } else {
219 // Just a simple swap will suffice
220 thesoup::types::Slice<T> slice {std::move(loader(page_number_to_insert_to).unwrap())};
221 page_table[page_table_entry].slice.start = slice.start;
222 page_table[page_table_entry].slice.size = slice.size;
223 page_table[page_table_entry].page_number = page_number_to_insert_to;
224 }
225
226 page_table[page_table_entry].valid = true;
227 page_table[page_table_entry].page_number = page_number_to_insert_to;
228 }
229
230
231 page_table[page_table_entry].slice.size++;
232 page_table[page_table_entry].slice[page_offset] = item;
233 _size++;
234 }
